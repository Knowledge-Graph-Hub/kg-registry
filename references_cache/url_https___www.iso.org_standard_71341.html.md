(function () {
    function getPageLanguage() {
        const lang = (document.documentElement.getAttribute('lang') || 'en')
            .toLowerCase()
            .split('-')[0];
        return ['en', 'fr', 'es', 'ru'].includes(lang) ? lang : 'en';
    }

    const PAGE_LANG = getPageLanguage();
    // ref and id are required for product mapping (eg AFNOR)
    const endpoint =
        `https://www.iso.org/bootiful-services/opt-out/ip/check?type=std&id=63787&ref=63787&lang=${PAGE_LANG}&guilang=${PAGE_LANG}`;

    const LS_KEY = 'iso_buybox_country_dismissed_at';
    const DISMISS_TTL_MS = 24 * 60 * 60 * 1000;
    // const OPTOUT_COUNTRIES = ['GB', 'FR', 'ES', 'NO'];
    const OPTOUT_COUNTRIES = ['GB'];

    const HIDDEN_CLASS = 'js-buybox-hidden';
    const INSERTED_CLASS = 'js-country-buybox-banner';

    const TRANSLATIONS = {
        en: {
            body: 'You are visiting from {country}. Customers with a billing address in {country} must purchase this item from the {member} store.',
            button: 'Buy from {member}',
            remove: 'Remove this notice',
            removeSuffix: " as my billing address isn't in {country}.",
            fallbackItem: 'Webstore notice'
        },
        fr: {
            body: 'Si votre requête provient de {country}, veuillez vous rendre sur la boutique {member} pour acheter cet article.',
            button: 'Acheter chez {member}',
            remove: 'Supprimer cet avertissement',
            removeSuffix: " car ma requête ne provient pas de {country}.",
            fallbackItem: 'Avertissement'
        },
        es: {
            body: 'Parece que se está conectando desde {country}. Visite la tienda de {member} para comprar este artículo.',
            button: 'Comprar en {member}',
            remove: 'Eliminar este aviso',
            removeSuffix: ' porque no me estoy conectando desde {country}.',
            fallbackItem: 'Aviso de la tienda en línea'
        },
        ru: {
            body: 'Похоже, что вы подключаетесь из {country}. Пожалуйста, посетите магазин {member}, чтобы приобрести эту публикацию.',
            button: 'Купить в {member}',
            remove: 'Убрать это уведомление',
            removeSuffix: ', так как я не подключаюсь из {country}.',
            fallbackItem: 'Уведомление интернет-магазина'
        }
    };

    function t() {
        return TRANSLATIONS[PAGE_LANG] || TRANSLATIONS.en;
    }

    function addUtmParams(url) {
        try {
            //const u = new URL(url, window.location.origin);
            const u = new URL(url);
            u.searchParams.set('utm_source', 'iso');
            u.searchParams.set('utm_medium', 'referral');
            u.searchParams.set('utm_campaign', 'iso_redirect');
            return u.toString();
        } catch (e) {
            return url;
        }
    }

    function formatTemplate(template, values) {
        return template.replace(/\{(\w+)\}/g, function (_, key) {
            return values[key] != null ? values[key] : '';
        });
    }

    function dismissedRecently() {
        try {
            const raw = localStorage.getItem(LS_KEY);
            if (!raw) return false;
            const ts = Number(raw);
            if (!Number.isFinite(ts)) return false;
            return (Date.now() - ts) < DISMISS_TTL_MS;
        } catch (e) {
            return false;
        }
    }

    function setDismissedNow() {
        try {
            localStorage.setItem(LS_KEY, String(Date.now()));
        } catch (e) {}
    }

    function unhideAllBuyboxes() {
        document.querySelectorAll('.card.buybox.' + HIDDEN_CLASS).forEach(function (el) {
            el.classList.remove(HIDDEN_CLASS);
            el.style.display = '';
            el.removeAttribute('aria-hidden');
        });
    }

    function removeAllInsertedBanners() {
        document.querySelectorAll('.' + INSERTED_CLASS).forEach(function (el) {
            el.remove();
        });
    }

    function resetToDefault() {
        removeAllInsertedBanners();
        unhideAllBuyboxes();
    }

    function createBannerCard(itemNumber, country, member, externalLink, countryIso) {
        const strings = t();

        const card = document.createElement('div');
        card.className = 'card ' + INSERTED_CLASS;

        const header = document.createElement('div');
        header.className = 'card-header text-center--OFF';

        const h4 = document.createElement('h4');
        h4.textContent = itemNumber;
        header.appendChild(h4);

        const body = document.createElement('div');
        body.className = 'card-body';

        const p = document.createElement('p');
        p.textContent = formatTemplate(strings.body, {
            country: country,
            member: member
        });

        const a = document.createElement('a');
        a.setAttribute('role', 'button');
        a.href = addUtmParams(externalLink);
        a.target = '_blank';
        a.rel = 'noopener noreferrer';
        a.className = 'btn btn-info btn-md';
        a.textContent = formatTemplate(strings.button, {
            member: member
        });

        // add dynamic country code
        if (countryIso) {
            a.dataset.storeloc = countryIso.toUpperCase();
        }

        body.appendChild(p);
        body.appendChild(a);

        const footer = document.createElement('div');
        footer.className = 'card-footer';

        const p2 = document.createElement('p');
        p2.className = 'text-xs';

        const removeLink = document.createElement('a');
        removeLink.className = 'removeBanner';
        removeLink.href = '#';
        removeLink.textContent = strings.remove;

        p2.appendChild(removeLink);
        p2.appendChild(document.createTextNode(formatTemplate(strings.removeSuffix, {
            country: country
        })));

        footer.appendChild(p2);

        card.appendChild(header);
        card.appendChild(body);
        card.appendChild(footer);

        return card;
    }

    function applyCountryBehavior(data) {
        if (dismissedRecently()) {
            resetToDefault();
            return;
        }

        const buyboxes = Array.from(document.querySelectorAll('.card.buybox'));
        if (!buyboxes.length) return;

        removeAllInsertedBanners();

        const strings = t();

        buyboxes.forEach(function (buybox) {
            buybox.classList.add(HIDDEN_CLASS);
            buybox.style.display = 'none';
            buybox.setAttribute('aria-hidden', 'true');

            const h4 = buybox.querySelector('.card-header > h4');
            const itemNumber = ((h4 && h4.textContent) || '').trim() || strings.fallbackItem;

            const banner = createBannerCard(
                itemNumber,
                data.country,
                data.member,
                data.redirectionUrl,
                data.country_iso
            );
            buybox.insertAdjacentElement('afterend', banner);
        });
    }

    function fetchAndApply() {
        if (dismissedRecently()) {
            resetToDefault();
            return;
        }

        fetch(endpoint)
            .then(function (res) {
                if (!res.ok) throw new Error('Network response was not ok');
                return res.json();
            })
            .then(function (data) {
                const countryIso = ((data && data.country_iso) || '').toUpperCase();
                const shouldShowBanner = (
                    OPTOUT_COUNTRIES.includes(countryIso) &&
                    data &&
                    data.country &&
                    data.member &&
                    data.redirectionUrl
                );

                if (shouldShowBanner) {
                    applyCountryBehavior(data);
                    return;
                }

                resetToDefault();
            })
            .catch(function (err) {
                console.error('IP check failed:', err);
                resetToDefault();
            });
    }

    document.addEventListener('click', function (e) {
        const link = e.target.closest('.removeBanner');
        if (!link) return;

        e.preventDefault();
        setDismissedNow();
        resetToDefault();
    });

    function init() {
        fetchAndApply();
    }

    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', init);
    } else {
        init();
    }
})();
</script>

          <div class="d-flex justify-content-between align-items-center"><h2>Abstract</h2></div>
            <div itemprop="description">
              <p><p>This document establishes a vocabulary for cross-domain resource description, known as the Dublin Core metadata terms (hereafter DCMI Metadata Terms). It includes all of the properties and classes in the main namespace of DCMI Metadata Terms[1] (hereafter "the /terms/ namespace"), as published in the DCMI Recommendation document "DCMI Metadata Terms" of 2012 (DCMI-TERMS and Annex A). As explained in Annex B, these properties and classes can be identified by URIs for use in linked data.</p>
<p>NOTE    The 15 terms of the original Dublin Core Metadata Element Set, as defined in the namespace https://purl.org/dc/elements/1.1/ (hereafter "the /elements/1.1/ namespace"), are also documented in the DCMI Recommendation "DCMI Metadata Terms" and in ISO 15836-1.</p>
<p>This document does not contain the following supporting terms from "DCMI Metadata Terms" specification:</p>
<p>a)   terms from the /elements/1.1/ namespace (included in ISO 15836-1);</p>
<p>b)   vocabulary encoding schemes;</p>
<p>c)   syntax encoding schemes;</p>
<p>d)   DCMI Type vocabulary;</p>
<p>e)   terms related to the DCMI Abstract Model.</p>
<p>Both ISO 15836-1 and this document include the 15 so-called core terms, but in ISO 15836-1 they are from the /elements/1.1/ namespace, and in this document from the /terms/ namespace. In the latter, the terms have narrower semantics due to formal domain and range specifications.</p>
<p>This document does not limit what might be a resource.</p>
<p>This document does not provide specific implementation guidelines. The properties and classes are typically used in the context of an application profile, which constrains or specifies their use in accordance with local or community-based requirements and policies.</p>

<p>[1] Available at: https://purl.org/dc/terms/.</p></p>
